client = openai.Client(api_key="{{api_key}}", base_url="http://148.187.108.173:8080")
res = client.chat.completions.create(
    model="[CHOOSE_MODEL_NAME]",
    messages=[

            "role": "user",
        }
    ],
    stream=True,
)

        print(chunk.choices[0].delta.content, end="", flush=True)
